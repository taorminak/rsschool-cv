                        const Cat = (function(name, weight) {
                          let cats = [];

                          function Cat(name, weight) {
                            if (!name || !weight) {
                              throw new Error('Both name and weight are required');
                            }
                            
                            Object.defineProperty(this, 'weight', {
                              get: function() {
                                return weight;
                              },
                              set: function(value) {
                                weight = value;
                              }
                            });
                            
                            this.name = name;
                            this.weight = weight;
                            cats.push(this);
                          }

                          Cat.averageWeight = function() {
                            let totalWeight = 0;
                            
                            if (cats.length === 0) {
                              return undefined;
                            }
                            
                            for (let i = 0; i < cats.length; i++) {
                              totalWeight += cats[i].weight;
                            }
                            
                            return totalWeight / cats.length;
                          };
                          
                          return Cat;
                        });
